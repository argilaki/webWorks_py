    function addNewsletter() {
        var email = $('#newsletter_email');
        var newsletter_birth_year = $('#newsletter_birth_year');
        var newsletter_birth_month = $('#newsletter_birth_month');
        var newsletter_birth_day = $('#newsletter_birth_day');

        if ($.trim(email.val()) == '') {
            $('#return_newsletter_message').html('<span class="text-danger">لطفا ایمیل را وارد کنید.</span>');
            return false;
        }

        $.post('https://hashure.com/addNewsLetter', {
            'email': email.val(),
            'birth_year': newsletter_birth_year.val(),
            'birth_month': newsletter_birth_month.val(),
            'birth_day': newsletter_birth_day.val(),
            '_token': '659Xkuc65nBKxYeZWChew7mjDMkpwXbOFVJk7hp8'
        }, function (data) {
            $('#return_newsletter_message').html('<span class="text-success">' + data['message'] + '</span>');
            $('#newsletter_form').hide();
        }).fail(function (data) {
            var all_errors = data.responseJSON;
            if (data.responseJSON && all_errors['errors']) {
                $.each(all_errors['errors'], function (i, row) {
                    $('#return_newsletter_message').append('<div class="alert alert-danger">' + row[0] + '</div>');
                });
            }
        });
    }
